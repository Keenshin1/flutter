Módulo 18
Navegação entre telas
Todo app real tem mais de uma tela. O Flutter usa uma pilha de rotas — você empurra uma nova tela por cima e volta removendo ela da pilha.
Navigator.push e Navigator.pop
O Navigator gerencia uma pilha de telas. push() coloca uma nova tela no topo; pop() remove a tela atual e volta para a anterior. O botão de voltar da AppBar chama pop() automaticamente.
Dart
import 'package:flutter/material.dart';

void main() => runApp(const MaterialApp(home: TelaInicial()));

class TelaInicial extends StatelessWidget {
  const TelaInicial({super.key});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('Tela Inicial')),
      body: Center(
        child: ElevatedButton(
          onPressed: () {
            Navigator.push(
              context,
              MaterialPageRoute(builder: (_) => const TelaDetalhe()),
            );
          },
          child: const Text('Ir para Detalhe'),
        ),
      ),
    );
  }
}

class TelaDetalhe extends StatelessWidget {
  const TelaDetalhe({super.key});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('Detalhe')),
      body: Center(
        child: ElevatedButton(
          onPressed: () => Navigator.pop(context),
          child: const Text('Voltar'),
        ),
      ),
    );
  }
}
MaterialPageRoute define a animação de transição entre telas. O parâmetro builder recebe o contexto e retorna o widget da nova tela.
Passando dados para outra tela
Para enviar dados a uma tela, basta passar argumentos pelo construtor do widget de destino — exatamente como em qualquer classe Dart.
Dart
class TelaProduto extends StatelessWidget {
  final String nome;
  final double preco;

  const TelaProduto({super.key, required this.nome, required this.preco});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text(nome)),
      body: Center(
        child: Text('R\$ $preco', style: const TextStyle(fontSize: 32)),
      ),
    );
  }
}

// Na tela que navega:
Navigator.push(
  context,
  MaterialPageRoute(
    builder: (_) => const TelaProduto(nome: 'Café', preco: 12.50),
  ),
);
Retornando dados com pop()
pop() pode devolver um valor para a tela anterior. Quem fez o push() recebe esse valor com await, pois a navegação é assíncrona.
Dart
// Tela que retorna um valor:
Navigator.pop(context, 'opcao_escolhida');

// Tela que aguarda o retorno:
final resultado = await Navigator.push<String>(
  context,
  MaterialPageRoute(builder: (_) => const TelaEscolha()),
);

if (resultado != null) {
  print('Usuário escolheu: $resultado');
}
O retorno de push() é Future<T?> — pode ser null se o usuário pressionar o botão de voltar sem chamar pop() explicitamente. Sempre verifique antes de usar.
Exercícios de fixação
01
Duas telas
push e pop básicos
Básico
Crie um app com duas telas: TelaHome e TelaSobre. A TelaHome tem um botão "Sobre o app" que navega para TelaSobre. A TelaSobre exibe um texto qualquer e tem um botão "Voltar" que chama Navigator.pop().
Solução
import 'package:flutter/material.dart';

void main() => runApp(const MaterialApp(home: TelaHome()));

class TelaHome extends StatelessWidget {
  const TelaHome({super.key});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('Home')),
      body: Center(
        child: ElevatedButton(
          onPressed: () => Navigator.push(
            context,
            MaterialPageRoute(builder: (_) => const TelaSobre()),
          ),
          child: const Text('Sobre o app'),
        ),
      ),
    );
  }
}

class TelaSobre extends StatelessWidget {
  const TelaSobre({super.key});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('Sobre')),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            const Text('App de exemplo — v1.0'),
            const SizedBox(height: 16),
            ElevatedButton(
              onPressed: () => Navigator.pop(context),
              child: const Text('Voltar'),
            ),
          ],
        ),
      ),
    );
  }
}
02
Lista de produtos
Passando dados pelo construtor
Médio
Crie uma TelaLista com uma ListView de 4 produtos (nome + preço). Ao tocar em um item, navegue para TelaDetalhe passando o nome e o preço pelo construtor. A TelaDetalhe exibe essas informações de forma destacada e tem o botão de voltar da AppBar (automático).
Solução
import 'package:flutter/material.dart';

void main() => runApp(const MaterialApp(home: TelaLista()));

class TelaLista extends StatelessWidget {
  const TelaLista({super.key});

  static const _produtos = [
    {'nome': 'Café',     'preco': 12.50},
    {'nome': 'Leite',    'preco': 5.90},
    {'nome': 'Pão',      'preco': 8.00},
    {'nome': 'Manteiga', 'preco': 18.00},
  ];

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('Produtos')),
      body: ListView.builder(
        itemCount: _produtos.length,
        itemBuilder: (context, i) {
          final p = _produtos[i];
          return ListTile(
            title: Text(p['nome']! as String),
            subtitle: Text('R\$ ${p['preco']}'),
            trailing: const Icon(Icons.chevron_right),
            onTap: () => Navigator.push(
              context,
              MaterialPageRoute(
                builder: (_) => TelaDetalhe(
                  nome: p['nome']! as String,
                  preco: p['preco']! as double,
                ),
              ),
            ),
          );
        },
      ),
    );
  }
}

class TelaDetalhe extends StatelessWidget {
  final String nome;
  final double preco;

  const TelaDetalhe({super.key, required this.nome, required this.preco});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text(nome)),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            Text(nome, style: const TextStyle(fontSize: 28, fontWeight: FontWeight.bold)),
            const SizedBox(height: 8),
            Text('R\$ $preco', style: const TextStyle(fontSize: 22)),
          ],
        ),
      ),
    );
  }
}
03
Formulário com retorno
pop() retornando dados
Desafio
Crie um app com duas telas. A TelaHome exibe um nome (inicialmente "Nenhum") e um botão "Editar nome". A TelaFormulario tem um TextField e um botão "Salvar" que chama Navigator.pop(context, nomeDigitado). De volta na TelaHome, use await Navigator.push() para receber o valor e atualizar o nome exibido com setState().
Solução
import 'package:flutter/material.dart';

void main() => runApp(const MaterialApp(home: TelaHome()));

class TelaHome extends StatefulWidget {
  const TelaHome({super.key});
  @override
  State<TelaHome> createState() => _TelaHomeState();
}

class _TelaHomeState extends State<TelaHome> {
  String _nome = 'Nenhum';

  Future<void> _editarNome() async {
    final resultado = await Navigator.push<String>(
      context,
      MaterialPageRoute(builder: (_) => const TelaFormulario()),
    );
    if (resultado != null && resultado.isNotEmpty) {
      setState(() => _nome = resultado);
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('Perfil')),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            Text('Nome: $_nome', style: const TextStyle(fontSize: 24)),
            const SizedBox(height: 16),
            ElevatedButton(
              onPressed: _editarNome,
              child: const Text('Editar nome'),
            ),
          ],
        ),
      ),
    );
  }
}

class TelaFormulario extends StatefulWidget {
  const TelaFormulario({super.key});
  @override
  State<TelaFormulario> createState() => _TelaFormularioState();
}

class _TelaFormularioState extends State<TelaFormulario> {
  final _controller = TextEditingController();

  @override
  void dispose() {
    _controller.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('Editar nome')),
      body: Padding(
        padding: const EdgeInsets.all(24),
        child: Column(
          children: [
            TextField(
              controller: _controller,
              decoration: const InputDecoration(labelText: 'Novo nome'),
            ),
            const SizedBox(height: 24),
            ElevatedButton(
              onPressed: () => Navigator.pop(context, _controller.text),
              child: const Text('Salvar'),
            ),
          ],
        ),
      ),
    );
  }
}
